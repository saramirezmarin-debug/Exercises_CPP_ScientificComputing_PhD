103 Xd.resize( X.size() );
104 Xd = X.template cast<Dual<T>>(); // convert from double to dual
105 dual_res = this->F( Xd );
106 res.resize(N);

108 res(i) = dual_res(i).value(); // convert from dual to double

112 // compute Jacobian JF(X) for the problem
113 d_mat JF( d_vec const & X ) const override {
114 d_dual_vec Xd, column;
115 d_mat JF;
116 integer N = X.rows();
117 JF.resize( N, N ); // Jacobian allocation
118 Xd.resize( N );
119 column.resize( N );
120 Xd = X.template cast<Dual<T>>(); // convert from double to dual
121 Dual<T> D(0,1); // 0 + 1 * epsilon
122 for ( integer i = 0; i < N; ++i ) {
123 // set x(i) = x(i) + epsilon --> compute derivative orespect to x(i)
124 Xd(i) += D;
125 column.noalias() = this->F( Xd );
126 Xd(i) -= D;
127 for ( integer j = 0; j < N; ++j ) {
128 JF( j, i ) = column(j).dual();
129 }
130 }

158 void
159 solve()
160 {
161 // initialize the loop
162 d_vec x0, x1, d0, d1, F0, F1;
163 d_mat JF0;
164 integer N = _problem->dim();
165 x0.resize(N);
166 x1.resize(N);
167 F0.resize(N);
168 F1.resize(N);
169 JF0.resize(N,N);
170 d0.resize(N);
171 d1.resize(N);
172
173 _converged = false;
174 x0 = _problem->initial_point();
175 for ( integer iter = 0; iter < _options.max_iter; ++iter )
176 {
177 F0 = _problem->F( x0 );
178 // check || F(x0) || for temination
179 T F0_norm = F0.template lpNorm<Eigen::Infinity>();
180
181 if (_options.verbose)
182 {
183 std::cout << iter << " ||F|| = " << F0_norm << '\n';
184 }
185 _converged = F0_norm <= _options.tolerance;
186 if ( _converged )
187 {
188 _x = x0;
189 return;
190 }
191
192 // compute direction
193 JF0 = _problem->JF( x0 );
194 d0 = -JF0.fullPivLu().solve(F0);
195 T d0norm2 = d0.dot(d0);
196 // damp loop
197 T alpha = 1;
198 bool ok = false;
199 for ( integer i = 0; !ok && i < _options.max_sub_iter; ++i ) {
200 x1 = x0 + alpha * d0;
201 // check Deufhard condition for x1
202 F1 = _problem->F( x1 );
203 d1 = -JF0.fullPivLu().solve(F1);
204 T d1norm2 = d1.dot(d1);
205 ok = d1norm2 <= d0norm2*(1-alpha/2);
206 if ( !ok ) alpha *= _options.damp_factor;
207 }
208 if ( !ok )
209 {
210 _x = x0;
211 return; // failed Newton
212 }
213 x0 = x1; // step accepted
214 }
215 _x = x0;
216 }
217 };
