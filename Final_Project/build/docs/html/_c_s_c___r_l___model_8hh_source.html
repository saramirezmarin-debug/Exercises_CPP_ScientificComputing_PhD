210 template <typename Vec>
211 Control<typename Vec::Scalar>
212 control(const CSC_RL_Parameters& p,
213 const Vec& x,
214 const Input<typename Vec::Scalar>& input) const
215 {
216 using T = typename Vec::Scalar;
217
218 Control<T> c;
219
220 const T ed = x(ED);
221 const T eq = x(EQ);
222 const T id = x(ID);
223 const T iq = x(IQ);
224 const T vd = x(VD);
225 const T vq = x(VQ);
226 const T istk = x(ISTK);
227
228 const T xi_pll = x(XI_PLL);
229 const T xi_ucd = x(XI_UCD);
230 const T xi_ucq = x(XI_UCQ);
231 const T xi_isd = x(XI_ISD);
232 const T xi_isq = x(XI_ISQ);
233 const T xi_idc2 = x(XI_IDC2);
234
235 // ------------------------------------------------------------
236 // PLL
237 // ------------------------------------------------------------
238 c.e_pll = eq / p.Vdq_nom;
239 c.w_hat = p.w0_pll + p.kp_pll * c.e_pll + p.ki_pll * xi_pll;
240
241 // ------------------------------------------------------------
242 // References
243 // ------------------------------------------------------------
244 c.idc_ref = input.idc_ref;
245 c.Qref = input.Q_ref;
246
247 // ------------------------------------------------------------
248 // Outer loop
249 // ------------------------------------------------------------
250 c.Eidc2 = c.idc_ref * c.idc_ref - istk * istk;
251 c.Pref = p.kpO * c.Eidc2 + p.kiO * xi_idc2;
252
253 // ------------------------------------------------------------
254 // AC current references
255 // ------------------------------------------------------------
256 const T V2 = ed * ed + eq * eq;
257
258 c.idr = (ed * c.Pref + eq * c.Qref) / V2;
259 c.iqr = (eq * c.Pref - ed * c.Qref) / V2;
260
261 c.Eid = c.idr - id;
262 c.Eiq = c.iqr - iq;
263
264 // ------------------------------------------------------------
265 // Current controller
266 // ------------------------------------------------------------
267 c.vdr = ed + c.w_hat * p.Lf * iq - p.Rf * id - p.kp2 * c.Eid - p.ki2 * xi_isd;
268 c.vqr = eq - c.w_hat * p.Lf * id - p.Rf * iq - p.kp2 * c.Eiq - p.ki2 * xi_isq;
269 c.Evd = c.vdr - vd;
270 c.Evq = c.vqr - vq;
271
272 // ------------------------------------------------------------
273 // Voltage controller
274 // ------------------------------------------------------------
275 c.md = (id + c.w_hat * p.Cf * vq - p.kp1 * c.Evd - p.ki1 * xi_ucd) / istk;
276 c.mq = (iq - c.w_hat * p.Cf * vd - p.kp1 * c.Evq - p.ki1 * xi_ucq) / istk;
277
278 return c;
279 }

422 template <typename Vec>
423 void rhs(const CSC_RL_Parameters& p,
424 ODE::real_type t,
425 const Vec& x,
426 Vec& dxdt,
427 const Input<typename Vec::Scalar>& input) const
428 {
429 using T = typename Vec::Scalar;
430
431 dxdt.resize(NSTATES);
432
433 const T igd = x(IGD);
434 const T igq = x(IGQ);
435 const T ed = x(ED);
436 const T eq = x(EQ);
437
438 const T id = x(ID);
439 const T iq = x(IQ);
440 const T vd = x(VD);
441 const T vq = x(VQ);
442 const T istk = x(ISTK);
443
444 const Control<T> c = control(p, x, input);
445
446 // ------------------------------------------------------------
447 // PCC dynamics
448 // ------------------------------------------------------------
449 dxdt(IGD) = (p.egd - p.Rg * igd - ed + c.w_hat * p.Lg * igq) / p.Lg;
450 dxdt(IGQ) = (p.egq - p.Rg * igq - eq - c.w_hat * p.Lg * igd) / p.Lg;
451
452 dxdt(ED) = (igd - id + c.w_hat * p.Cg * eq) / p.Cg;
453 dxdt(EQ) = (igq - iq - c.w_hat * p.Cg * ed) / p.Cg;
454
455 // ------------------------------------------------------------
456 // CSC dynamics
457 // ------------------------------------------------------------
458 dxdt(ID) = (ed - vd - p.Rf * id + c.w_hat * p.Lf * iq) / p.Lf;
459 dxdt(IQ) = (eq - vq - p.Rf * iq - c.w_hat * p.Lf * id) / p.Lf;
460
461 dxdt(VD) = (id - c.md * istk + c.w_hat * p.Cf * vq) / p.Cf;
462 dxdt(VQ) = (iq - c.mq * istk - c.w_hat * p.Cf * vd) / p.Cf;
463
464 dxdt(ISTK) = (c.md * vd + c.mq * vq - p.RL * istk) / p.Ldc;
465
466 // ------------------------------------------------------------
467 // PLL
468 // ------------------------------------------------------------
469 dxdt(THETA_HAT) = c.w_hat;
470 dxdt(XI_PLL) = c.e_pll;
471
472 // ------------------------------------------------------------
473 // Integrators
474 // ------------------------------------------------------------
475 dxdt(XI_UCD) = c.Evd;
476 dxdt(XI_UCQ) = c.Evq;
477 dxdt(XI_ISD) = c.Eid;
478 dxdt(XI_ISQ) = c.Eiq;
479 dxdt(XI_IDC2) = c.Eidc2;
480 }

575inline ODE::vec_type make_csc_rl_equilibrium_initial_guess(const CSC_RL_Parameters& p,
576 const EquilibriumReferences& ref)
577{
578 using namespace CSC_RL_Equilibrium_Detail;
579
580 ODE::vec_type z0;
581 z0.resize(Z_N);
582 z0.setZero();
583
584 const ODE::real_type ed0 = p.Vdq_nom;
585 const ODE::real_type eq0 = 0.0;
586
587 const ODE::real_type P0 = p.RL * ref.idc_ref * ref.idc_ref;
588
589 const ODE::real_type id0 = P0 / ed0;
590 const ODE::real_type iq0 = -ref.Q_ref / ed0;
591
592 z0(Z_IGD) = id0;
593 z0(Z_IGQ) = iq0;
594
595 z0(Z_ED) = ed0;
596 z0(Z_EQ) = eq0;
597
598 z0(Z_ID) = id0;
599 z0(Z_IQ) = iq0;
600
601 z0(Z_VD) = ed0;
602 z0(Z_VQ) = eq0;
603
604 z0(Z_ISTK) = ref.idc_ref;
605
606 z0(Z_XI_PLL) = 0.0;
607 z0(Z_XI_UCD) = 0.0;
608 z0(Z_XI_UCQ) = 0.0;
609 z0(Z_XI_ISD) = 0.0;
610 z0(Z_XI_ISQ) = 0.0;
611 z0(Z_XI_IDC2) = 0.0;
612
613 return z0;
614}

698 d_dual_vec F(d_dual_vec const& X) const override
699 {
700 using DualReal = AD::Dual<real>;
701 using namespace CSC_RL_Equilibrium_Detail;
702
703 d_dual_vec x;
704 d_dual_vec dxdt;
705 d_dual_vec res;
706
707 x.resize(NSTATES);
708 dxdt.resize(NSTATES);
709 res.resize(Z_N);
710
711 x.setZero();
712 dxdt.setZero();
713 res.setZero();
714
715 // ------------------------------------------------------------
716 // Reduced equilibrium vector -> full CSC_RL state vector
717 // ------------------------------------------------------------
718 x(IGD) = X(Z_IGD);
719 x(IGQ) = X(Z_IGQ);
720 x(ED) = X(Z_ED);
721 x(EQ) = X(Z_EQ);
722
723 x(ID) = X(Z_ID);
724 x(IQ) = X(Z_IQ);
725 x(VD) = X(Z_VD);
726 x(VQ) = X(Z_VQ);
727 x(ISTK) = X(Z_ISTK);
728
729 // theta_hat is not part of the equilibrium unknowns.
730 x(THETA_HAT) = DualReal(p_.x0(THETA_HAT));
731
732 x(XI_PLL) = X(Z_XI_PLL);
733 x(XI_UCD) = X(Z_XI_UCD);
734 x(XI_UCQ) = X(Z_XI_UCQ);
735 x(XI_ISD) = X(Z_XI_ISD);
736 x(XI_ISQ) = X(Z_XI_ISQ);
737 x(XI_IDC2) = X(Z_XI_IDC2);
738
739 // ------------------------------------------------------------
740 // Fixed references for equilibrium
741 // ------------------------------------------------------------
742 typename CSC_RL_Model::template Input<DualReal> input;
743
744 input.idc_ref = DualReal(ref_.idc_ref);
745 input.Q_ref = DualReal(ref_.Q_ref);
746
747 // ------------------------------------------------------------
748 // Evaluate the same model used by ODE4
749 // ------------------------------------------------------------
750 model_.rhs(p_, p_.t0, x, dxdt, input);
751
752 // ------------------------------------------------------------
753 // Equilibrium residuals: dx/dt = 0
754 // theta_hat is excluded because dtheta_hat/dt = omega_hat.
755 // ------------------------------------------------------------
756 res(Z_IGD) = dxdt(IGD);
757 res(Z_IGQ) = dxdt(IGQ);
758 res(Z_ED) = dxdt(ED);
759 res(Z_EQ) = dxdt(EQ);
760
761 res(Z_ID) = dxdt(ID);
762 res(Z_IQ) = dxdt(IQ);
763 res(Z_VD) = dxdt(VD);
764 res(Z_VQ) = dxdt(VQ);
765 res(Z_ISTK) = dxdt(ISTK);
766
767 res(Z_XI_PLL) = dxdt(XI_PLL);
768 res(Z_XI_UCD) = dxdt(XI_UCD);
769 res(Z_XI_UCQ) = dxdt(XI_UCQ);
770 res(Z_XI_ISD) = dxdt(XI_ISD);
771 res(Z_XI_ISQ) = dxdt(XI_ISQ);
772 res(Z_XI_IDC2) = dxdt(XI_IDC2);
773
774 return res;
775 }

854inline bool compute_csc_rl_equilibrium(const CSC_RL_Model& model,
855 CSC_RL_Parameters& p,
856 const EquilibriumOptions& options)
857{
858 EquilibriumReferences eq_ref;
859
860 if (options.use_reference_override)
861 {
862 eq_ref = options.reference;
863 }
864 else
865 {
866 eq_ref.idc_ref = p.idc_ref.value(p.t0);
867 eq_ref.Q_ref = p.Q_ref.value(p.t0);
868 }
869
870 if (options.verbose)
871 {
872 std::cout << "Computing CSC_RL equilibrium point...\n";
873 std::cout << "idc_ref = " << eq_ref.idc_ref << " A\n";
874 std::cout << "Q_ref = " << eq_ref.Q_ref << " var\n\n";
875 }
876
877 CSC_RL_EquilibriumProblem eq_problem(model, p, eq_ref);
878
879 ODE::vec_type z_star;
880 const bool ok = solve_equilibrium_problem(eq_problem, options, z_star);
881
882 if (!ok)
883 {
884 return false;
885 }
886
887 const ODE::vec_type x0_eq =
888 expand_csc_rl_equilibrium_to_full_state(z_star, p);
889
890 apply_csc_rl_equilibrium_to_parameters(p, x0_eq);
891
892 return true;
893}
