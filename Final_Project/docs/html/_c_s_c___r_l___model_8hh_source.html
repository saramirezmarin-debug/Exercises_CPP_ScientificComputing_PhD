59 template <typename Vec>
60 Control<typename Vec::Scalar>
61 control(const CSC_RL_Parameters& p,
62 const Vec& x,
63 const Input<typename Vec::Scalar>& input) const
64 {
65 using T = typename Vec::Scalar;
66
67 Control<T> c;
68
69 const T ed = x(ED);
70 const T eq = x(EQ);
71 const T id = x(ID);
72 const T iq = x(IQ);
73 const T vd = x(VD);
74 const T vq = x(VQ);
75 const T istk = x(ISTK);
76
77 const T xi_pll = x(XI_PLL);
78 const T xi_ucd = x(XI_UCD);
79 const T xi_ucq = x(XI_UCQ);
80 const T xi_isd = x(XI_ISD);
81 const T xi_isq = x(XI_ISQ);
82 const T xi_idc2 = x(XI_IDC2);
83
84 // ------------------------------------------------------------
85 // PLL
86 // ------------------------------------------------------------
87 c.e_pll = eq / p.Vdq_nom;
88 c.w_hat = p.w0_pll + p.kp_pll * c.e_pll + p.ki_pll * xi_pll;
89
90 // ------------------------------------------------------------
91 // References
92 // ------------------------------------------------------------
93 c.idc_ref = input.idc_ref;
94 c.Qref = input.Q_ref;
95
96 // ------------------------------------------------------------
97 // Outer loop
98 // ------------------------------------------------------------
99 c.Eidc2 = c.idc_ref * c.idc_ref - istk * istk;
100 c.Pref = p.kpO * c.Eidc2 + p.kiO * xi_idc2;
101
102 // ------------------------------------------------------------
103 // AC current references
104 // ------------------------------------------------------------
105 const T V2 = ed * ed + eq * eq;
106
107 c.idr = (ed * c.Pref + eq * c.Qref) / V2;
108 c.iqr = (eq * c.Pref - ed * c.Qref) / V2;
109
110 c.Eid = c.idr - id;
111 c.Eiq = c.iqr - iq;
112
113 // ------------------------------------------------------------
114 // Current controller
115 // ------------------------------------------------------------
116 c.vdr = ed + c.w_hat * p.Lf * iq - p.Rf * id - p.kp2 * c.Eid - p.ki2 * xi_isd;
117 c.vqr = eq - c.w_hat * p.Lf * id - p.Rf * iq - p.kp2 * c.Eiq - p.ki2 * xi_isq;
118 c.Evd = c.vdr - vd;
119 c.Evq = c.vqr - vq;
120
121 // ------------------------------------------------------------
122 // Voltage controller
123 // ------------------------------------------------------------
124 c.md = (id + c.w_hat * p.Cf * vq - p.kp1 * c.Evd - p.ki1 * xi_ucd) / istk;
125 c.mq = (iq - c.w_hat * p.Cf * vd - p.kp1 * c.Evq - p.ki1 * xi_ucq) / istk;
126
127 return c;
128 }

130 template <typename Vec>
131 void rhs(const CSC_RL_Parameters& p,
132 ODE::real_type t,
133 const Vec& x,
134 Vec& dxdt,
135 const Input<typename Vec::Scalar>& input) const
136 {
137 using T = typename Vec::Scalar;
138
139 dxdt.resize(NSTATES);
140
141 const T igd = x(IGD);
142 const T igq = x(IGQ);
143 const T ed = x(ED);
144 const T eq = x(EQ);
145
146 const T id = x(ID);
147 const T iq = x(IQ);
148 const T vd = x(VD);
149 const T vq = x(VQ);
150 const T istk = x(ISTK);
151
152 const Control<T> c = control(p, x, input);
153
154 // ------------------------------------------------------------
155 // PCC dynamics
156 // ------------------------------------------------------------
157 dxdt(IGD) = (p.egd - p.Rg * igd - ed + c.w_hat * p.Lg * igq) / p.Lg;
158 dxdt(IGQ) = (p.egq - p.Rg * igq - eq - c.w_hat * p.Lg * igd) / p.Lg;
159
160 dxdt(ED) = (igd - id + c.w_hat * p.Cg * eq) / p.Cg;
161 dxdt(EQ) = (igq - iq - c.w_hat * p.Cg * ed) / p.Cg;
162
163 // ------------------------------------------------------------
164 // CSC dynamics
165 // ------------------------------------------------------------
166 dxdt(ID) = (ed - vd - p.Rf * id + c.w_hat * p.Lf * iq) / p.Lf;
167 dxdt(IQ) = (eq - vq - p.Rf * iq - c.w_hat * p.Lf * id) / p.Lf;
168
169 dxdt(VD) = (id - c.md * istk + c.w_hat * p.Cf * vq) / p.Cf;
170 dxdt(VQ) = (iq - c.mq * istk - c.w_hat * p.Cf * vd) / p.Cf;
171
172 dxdt(ISTK) = (c.md * vd + c.mq * vq - p.RL * istk) / p.Ldc;
173
174 // ------------------------------------------------------------
175 // PLL
176 // ------------------------------------------------------------
177 dxdt(THETA_HAT) = c.w_hat;
178 dxdt(XI_PLL) = c.e_pll;
179
180 // ------------------------------------------------------------
181 // Integrators
182 // ------------------------------------------------------------
183 dxdt(XI_UCD) = c.Evd;
184 dxdt(XI_UCQ) = c.Evq;
185 dxdt(XI_ISD) = c.Eid;
186 dxdt(XI_ISQ) = c.Eiq;
187 dxdt(XI_IDC2) = c.Eidc2;
188 }
