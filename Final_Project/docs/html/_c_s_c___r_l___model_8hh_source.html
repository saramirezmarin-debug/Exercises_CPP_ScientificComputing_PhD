61 template <typename Vec>
62 Control<typename Vec::Scalar>
63 control(const CSC_RL_Parameters& p,
64 const Vec& x,
65 const Input<typename Vec::Scalar>& input) const
66 {
67 using T = typename Vec::Scalar;
68
69 Control<T> c;
70
71 const T ed = x(ED);
72 const T eq = x(EQ);
73 const T id = x(ID);
74 const T iq = x(IQ);
75 const T vd = x(VD);
76 const T vq = x(VQ);
77 const T istk = x(ISTK);
78
79 const T xi_pll = x(XI_PLL);
80 const T xi_ucd = x(XI_UCD);
81 const T xi_ucq = x(XI_UCQ);
82 const T xi_isd = x(XI_ISD);
83 const T xi_isq = x(XI_ISQ);
84 const T xi_idc2 = x(XI_IDC2);
85
86 // ------------------------------------------------------------
87 // PLL
88 // ------------------------------------------------------------
89 c.e_pll = eq / p.Vdq_nom;
90 c.w_hat = p.w0_pll + p.kp_pll * c.e_pll + p.ki_pll * xi_pll;
91
92 // ------------------------------------------------------------
93 // References
94 // ------------------------------------------------------------
95 c.idc_ref = input.idc_ref;
96 c.Qref = input.Q_ref;
97
98 // ------------------------------------------------------------
99 // Outer loop
100 // ------------------------------------------------------------
101 c.Eidc2 = c.idc_ref * c.idc_ref - istk * istk;
102 c.Pref = p.kpO * c.Eidc2 + p.kiO * xi_idc2;
103
104 // ------------------------------------------------------------
105 // AC current references
106 // ------------------------------------------------------------
107 const T V2 = ed * ed + eq * eq;
108
109 c.idr = (ed * c.Pref + eq * c.Qref) / V2;
110 c.iqr = (eq * c.Pref - ed * c.Qref) / V2;
111
112 c.Eid = c.idr - id;
113 c.Eiq = c.iqr - iq;
114
115 // ------------------------------------------------------------
116 // Current controller
117 // ------------------------------------------------------------
118 c.vdr = ed + c.w_hat * p.Lf * iq - p.Rf * id - p.kp2 * c.Eid - p.ki2 * xi_isd;
119 c.vqr = eq - c.w_hat * p.Lf * id - p.Rf * iq - p.kp2 * c.Eiq - p.ki2 * xi_isq;
120 c.Evd = c.vdr - vd;
121 c.Evq = c.vqr - vq;
122
123 // ------------------------------------------------------------
124 // Voltage controller
125 // ------------------------------------------------------------
126 c.md = (id + c.w_hat * p.Cf * vq - p.kp1 * c.Evd - p.ki1 * xi_ucd) / istk;
127 c.mq = (iq - c.w_hat * p.Cf * vd - p.kp1 * c.Evq - p.ki1 * xi_ucq) / istk;
128
129 return c;
130 }

132 template <typename Vec>
133 void rhs(const CSC_RL_Parameters& p,
134 ODE::real_type t,
135 const Vec& x,
136 Vec& dxdt,
137 const Input<typename Vec::Scalar>& input) const
138 {
139 using T = typename Vec::Scalar;
140
141 dxdt.resize(NSTATES);
142
143 const T igd = x(IGD);
144 const T igq = x(IGQ);
145 const T ed = x(ED);
146 const T eq = x(EQ);
147
148 const T id = x(ID);
149 const T iq = x(IQ);
150 const T vd = x(VD);
151 const T vq = x(VQ);
152 const T istk = x(ISTK);
153
154 const Control<T> c = control(p, x, input);
155
156 // ------------------------------------------------------------
157 // PCC dynamics
158 // ------------------------------------------------------------
159 dxdt(IGD) = (p.egd - p.Rg * igd - ed + c.w_hat * p.Lg * igq) / p.Lg;
160 dxdt(IGQ) = (p.egq - p.Rg * igq - eq - c.w_hat * p.Lg * igd) / p.Lg;
161
162 dxdt(ED) = (igd - id + c.w_hat * p.Cg * eq) / p.Cg;
163 dxdt(EQ) = (igq - iq - c.w_hat * p.Cg * ed) / p.Cg;
164
165 // ------------------------------------------------------------
166 // CSC dynamics
167 // ------------------------------------------------------------
168 dxdt(ID) = (ed - vd - p.Rf * id + c.w_hat * p.Lf * iq) / p.Lf;
169 dxdt(IQ) = (eq - vq - p.Rf * iq - c.w_hat * p.Lf * id) / p.Lf;
170
171 dxdt(VD) = (id - c.md * istk + c.w_hat * p.Cf * vq) / p.Cf;
172 dxdt(VQ) = (iq - c.mq * istk - c.w_hat * p.Cf * vd) / p.Cf;
173
174 dxdt(ISTK) = (c.md * vd + c.mq * vq - p.RL * istk) / p.Ldc;
175
176 // ------------------------------------------------------------
177 // PLL
178 // ------------------------------------------------------------
179 dxdt(THETA_HAT) = c.w_hat;
180 dxdt(XI_PLL) = c.e_pll;
181
182 // ------------------------------------------------------------
183 // Integrators
184 // ------------------------------------------------------------
185 dxdt(XI_UCD) = c.Evd;
186 dxdt(XI_UCQ) = c.Evq;
187 dxdt(XI_ISD) = c.Eid;
188 dxdt(XI_ISQ) = c.Eiq;
189 dxdt(XI_IDC2) = c.Eidc2;
190 }

226inline ODE::vec_type make_csc_rl_equilibrium_initial_guess(const CSC_RL_Parameters& p,
227 const EquilibriumReferences& ref)
228{
229 using namespace CSC_RL_Equilibrium_Detail;
230
231 ODE::vec_type z0;
232 z0.resize(Z_N);
233 z0.setZero();
234
235 const ODE::real_type ed0 = p.Vdq_nom;
236 const ODE::real_type eq0 = 0.0;
237
238 const ODE::real_type P0 = p.RL * ref.idc_ref * ref.idc_ref;
239
240 const ODE::real_type id0 = P0 / ed0;
241 const ODE::real_type iq0 = -ref.Q_ref / ed0;
242
243 z0(Z_IGD) = id0;
244 z0(Z_IGQ) = iq0;
245
246 z0(Z_ED) = ed0;
247 z0(Z_EQ) = eq0;
248
249 z0(Z_ID) = id0;
250 z0(Z_IQ) = iq0;
251
252 z0(Z_VD) = ed0;
253 z0(Z_VQ) = eq0;
254
255 z0(Z_ISTK) = ref.idc_ref;
256
257 z0(Z_XI_PLL) = 0.0;
258 z0(Z_XI_UCD) = 0.0;
259 z0(Z_XI_UCQ) = 0.0;
260 z0(Z_XI_ISD) = 0.0;
261 z0(Z_XI_ISQ) = 0.0;
262 z0(Z_XI_IDC2) = 0.0;
263
264 return z0;

423inline bool compute_csc_rl_equilibrium(const CSC_RL_Model& model,
424 CSC_RL_Parameters& p,
425 const EquilibriumOptions& options)
426{
427 EquilibriumReferences eq_ref;
428
429 if (options.use_reference_override)
430 {
431 eq_ref = options.reference;
432 }
433 else
434 {
435 eq_ref.idc_ref = p.idc_ref.value(p.t0);
436 eq_ref.Q_ref = p.Q_ref.value(p.t0);
437 }
438
439 if (options.verbose)
440 {
441 std::cout << "Computing CSC_RL equilibrium point...\n";
442 std::cout << "idc_ref = " << eq_ref.idc_ref << " A\n";
443 std::cout << "Q_ref = " << eq_ref.Q_ref << " var\n";
444 }
445
446 CSC_RL_EquilibriumProblem eq_problem(model, p, eq_ref);
447
448 ODE::vec_type z_star;
449 const bool ok = solve_equilibrium_problem(eq_problem, options, z_star);
450
451 if (!ok)
452 {
453 return false;
454 }
455
456 const ODE::vec_type x0_eq =
457 expand_csc_rl_equilibrium_to_full_state(z_star, p);
458
459 apply_csc_rl_equilibrium_to_parameters(p, x0_eq);
460
461 return true;
462}
