11// ============================================================
12// Generic stair signal
13//
14//
15// Example:
16//
17// times = {0.0, 1.0, 2.0, 3.0}
18// values = {16.0, 50.0, 100.0, 80.0}
19//
20// Then:
21//
22// 0.0 <= t < 1.0 -> 16.0
23// 1.0 <= t < 2.0 -> 50.0
24// 2.0 <= t < 3.0 -> 100.0
25// t >= 3.0 -> 80.0
26//
27// If times and values are empty, the signal uses constant_value.
28// ============================================================
29struct StairSignal
30{
31
32 ODE::real_type constant_value = 0.0; // Default value if no stair is provided
33 std::vector<ODE::real_type> times; // Switching times of the stair signal
34 std::vector<ODE::real_type> values; // Values of the stair signal at the corresponding times
35
36 ODE::real_type value(ODE::real_type t) const
37 {
38 if (times.empty())
39 {
40 return constant_value;
41 }
42
43 if (t < times.front())
44 {
45 return values.front();
46 }
47
48 auto it = std::upper_bound(times.begin(), times.end(), t);
49
50 const std::size_t index =
51 static_cast<std::size_t>(std::distance(times.begin(), it) - 1);
52
53 return values[index];
54 }
55
56 void validate(const std::string& name) const
57 {
58 // Empty vectors are allowed.
59 // In that case the signal behaves as a constant.
60 if (times.empty() && values.empty())
61 {
62 return;
63 }
64
65 // It is invalid to define only times or only values.
66 if (times.empty() || values.empty())
67 {
68 throw std::runtime_error(name + ": times and values must both be empty or both be non-empty");
69 }
70
71 // Each switching time must have one associated value.
72 if (times.size() != values.size())
73 {
74 throw std::runtime_error(name + ": times and values must have the same size");
75 }
76
77 // The stair logic assumes that times are sorted.
78 if (!std::is_sorted(times.begin(), times.end()))
79 {
80 throw std::runtime_error(name + ": times must be sorted in ascending order");
81 }
82 }
83};
