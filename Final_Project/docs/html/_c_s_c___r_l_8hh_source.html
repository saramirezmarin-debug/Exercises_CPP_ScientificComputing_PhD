5// ============================================================
6// Generic stair signal
7//
8//
9// Example:
10//
11// times = {0.0, 1.0, 2.0, 3.0}
12// values = {16.0, 50.0, 100.0, 80.0}
13//
14// Then:
15//
16// 0.0 <= t < 1.0 -> 16.0
17// 1.0 <= t < 2.0 -> 50.0
18// 2.0 <= t < 3.0 -> 100.0
19// t >= 3.0 -> 80.0
20//
21// If times and values are empty, the signal uses constant_value.
22// ============================================================
23struct StairSignal
24{
25
26 ODE::real_type constant_value = 0.0; // Default value if no stair is provided
27 std::vector<ODE::real_type> times; // Switching times of the stair signal
28 std::vector<ODE::real_type> values; // Values of the stair signal at the corresponding times
29
30 ODE::real_type value(ODE::real_type t) const
31 {
32 if (times.empty() && values.empty())
33 {
34 return constant_value; // Return constant value if no stair is defined
35 }
36
37 if (times.empty() || values.empty())
38 {
39 throw std::runtime_error("StairSignal: times and values must both be non-empty or both empty");
40 }
41
42 // Ensure times and values have the same size
43 if (times.size() != values.size())
44 {
45 throw std::runtime_error("StairSignal: times and values must have the same size");
46 }
47
48 // If t is before the first switching time,
49 // use the first defined value.
50 if (t < times.front())
51 {
52 return values.front();
53 }
54
55 // upper_bound gives the first switching time strictly greater than t.
56 // The active value is therefore one index before that.
57 auto it = std::upper_bound(times.begin(), times.end(), t);
58
59 const std::size_t index =
60 static_cast<std::size_t>(std::distance(times.begin(), it) - 1);
61
62 return values[index];
63 }
64
65 void validate(const std::string& name) const
66 {
67 // Empty vectors are allowed.
68 // In that case the signal behaves as a constant.
69 if (times.empty() && values.empty())
70 {
71 return;
72 }
73
74 // It is invalid to define only times or only values.
75 if (times.empty() || values.empty())
76 {
77 throw std::runtime_error(name + ": times and values must both be empty or both be non-empty");
78 }
79
80 // Each switching time must have one associated value.
81 if (times.size() != values.size())
82 {
83 throw std::runtime_error(name + ": times and values must have the same size");
84 }
85
86 // The stair logic assumes that times are sorted.
87 if (!std::is_sorted(times.begin(), times.end()))
88 {
89 throw std::runtime_error(name + ": times must be sorted in ascending order");
90 }
91 }
92};
